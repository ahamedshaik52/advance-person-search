                              - btnSearch:
                                  Control: FluentV8/Button@1.5.13
                                  Properties:
                                    Height: |
                                      =36
                                    OnSelect: |-
                                      =/* =========================================================
                                         RESET STATE
                                         ========================================================= */
                                      Set(varAPS_Error,    "");
                                      Set(varNoResultsMessage, "");
                                      Set(varIsSearching,  true);
                                      Set(varCanSearch,    true);
                                      Set(varIsValidDOB,   true);
                                      Set(varSearchClicked, true);
                                      Set(varMatchMode, If(tglExactMatch.Checked, "Exact", "FuzzyPhonetic"));
                                      Clear(colAPS_Results);
                                      Clear(colAPS_Results_Flat);
                                      Clear(colPhoneNums);
                                      Clear(colAddressNums);
                                      Set(CurrentPage, 1);
                                      
                                      /* =========================================================
                                         NORMALIZE INPUTS
                                         ========================================================= */
                                      Set(var_fn,       Trim(txtFirstName.Value));
                                      Set(var_ln,       Trim(txtLastName.Value));
                                      Set(var_dobRaw,   Trim(txtDOB.Value));
                                      Set(var_ssnRaw,   Trim(txtSSN.Value));
                                      Set(var_pidRaw,   Trim(TextBox5.Value));
                                      Set(var_phoneRaw, Trim(txtPhoneNUmber.Value));
                                      
                                      Set(var_addr1,    Lower(Trim(txtAddress1.Value)));
                                      Set(var_addr2,    Lower(Trim(txtAddress2.Value)));
                                      Set(var_city,     Lower(Trim(TextBox10.Value)));
                                      Set(var_state,    Lower(Trim(TextBox11.Value)));
                                      Set(var_zip,      Lower(Trim(TextBox12.Value)));
                                      Set(var_county,   Lower(Trim(txtCounty.Value)));
                                      
                                      Set(var_gender, cmbGender.Selected);
                                      Set(var_race,   cmbRace.SelectedItems);
                                      
                                      Set(
                                          varSearchRequest,
                                          {
                                              matchMode: varMatchMode,
                                              firstName: var_fn,
                                              lastName: var_ln,
                                              dob: var_dobRaw,
                                              ssn: var_ssnRaw,
                                              personId: var_pidRaw,
                                              phone: var_phoneRaw,
                                              addressLine1: var_addr1,
                                              addressLine2: var_addr2,
                                              city: var_city,
                                              state: var_state,
                                              zipCode: var_zip,
                                              county: var_county,
                                              gender: Coalesce(var_gender.Value, ""),
                                              race: Concat(var_race, Value, ","),
                                              singleLineQuery: ""
                                          }
                                      );
                                      
                                      
                                      /* =========================================================
                                         DERIVED VALUES
                                         ========================================================= */
                                      Set(var_pid,         If(!IsBlank(var_pidRaw), Value(var_pidRaw), Blank()));
                                      Set(var_phoneDigits, Concat(MatchAll(var_phoneRaw, "\d"), FullMatch));
                                      Set(var_ssn,         Substitute(Substitute(var_ssnRaw, "-", ""), " ", ""));
                                      Set(var_dobClean,    Substitute(var_dobRaw, " ", ""));
                                      
                                      
                                      /* =========================================================
                                         RULE 1 — PRESENCE CHECK
                                         ========================================================= */
                                      If(
                                          varCanSearch             &&
                                          IsBlank(var_fn)          &&
                                          IsBlank(var_ln)          &&
                                          IsBlank(var_dobRaw)      &&
                                          IsBlank(var_ssnRaw)      &&
                                          IsBlank(var_pidRaw)      &&
                                          IsBlank(var_phoneRaw)    &&
                                          IsBlank(var_addr1)       &&
                                          IsBlank(var_addr2)       &&
                                          IsBlank(var_city)        &&
                                          IsBlank(var_state)       &&
                                          IsBlank(var_zip)         &&
                                          IsBlank(var_county)      &&
                                          IsBlank(var_gender.Value)&&
                                          CountRows(var_race) = 0,
                                          Set(varAPS_Error, "Enter at least one valid search identifier.");
                                          Set(varCanSearch, false)
                                      );
                                      
                                      
                                      /* =========================================================
                                         FIELD-LEVEL VALIDATIONS
                                         ========================================================= */
                                      If(
                                          varCanSearch && !IsBlank(var_fn) &&
                                          !IsMatch(var_fn, "^[A-Za-z' -]+$"),
                                          Set(varAPS_Error, "Invalid First Name.");
                                          Set(varCanSearch, false)
                                      );
                                      If(
                                          varCanSearch && !IsBlank(var_ln) &&
                                          !IsMatch(var_ln, "^[A-Za-z' -]+$"),
                                          Set(varAPS_Error, "Invalid Last Name.");
                                          Set(varCanSearch, false)
                                      );
                                      If(
                                          varCanSearch && !IsBlank(var_pidRaw) &&
                                          (!IsMatch(var_pidRaw, "^\d+$") || Len(var_pidRaw) < 2),
                                          Set(varAPS_Error, "Person ID must be numeric and at least 2 digits.");
                                          Set(varCanSearch, false)
                                      );
                                      If(
                                          varCanSearch && !IsBlank(var_phoneRaw) &&
                                          !IsMatch(var_phoneRaw, "^[\d\s\(\)\-\+\.x]+$"),
                                          Set(varAPS_Error, "Invalid Phone Number.");
                                          Set(varCanSearch, false)
                                      );
                                      If(
                                          varCanSearch && !IsBlank(var_ssnRaw) &&
                                          !(Len(var_ssn) = 9 || Len(var_ssn) = 4),
                                          Set(varAPS_Error, "SSN must be 9 or last 4 digits.");
                                          Set(varCanSearch, false)
                                      );
                                      
                                      
                                      /* =========================================================
                                         DOB VALIDATION
                                         ========================================================= */
                                      Set(
                                          varIsValidDOB,
                                          IsBlank(var_dobRaw) ||
                                          IsMatch(var_dobClean, "^\d{4}$") ||
                                          IsMatch(var_dobClean, "^(0?[1-9]|1[0-2])/\d{4}$") ||
                                          (
                                              IsMatch(var_dobClean, "^(0?[1-9]|1[0-2])/(0?[1-9]|[12]\d|3[01])/\d{4}$") &&
                                              With(
                                                  { dobParts: Split(var_dobClean, "/") },
                                                  With(
                                                      {
                                                          m: Value(First(dobParts).Value),
                                                          d: Value(Index(dobParts, 2).Value),
                                                          y: Value(Last(dobParts).Value)
                                                      },
                                                      Month(Date(y, m, d)) = m &&
                                                      Day(Date(y, m, d))   = d
                                                  )
                                              )
                                          )
                                      );
                                      If(
                                          varCanSearch && !varIsValidDOB,
                                          Set(varAPS_Error, "DOB must be MM/DD/YYYY, MM/YYYY, or YYYY.");
                                          Set(varCanSearch, false)
                                      );
                                      
                                      
                                      /* =========================================================
                                         MAIN SEARCH
                                         ========================================================= */
                                      If(
                                          varCanSearch,
                                          IfError(
                                              Set(
                                                  SearchResults,
                                                  PersonSearchDataFlow.Run(
                                                      JSON(varSearchRequest, JSONFormat.Compact)
                                                  ).results
                                              );
                                              ClearCollect(
                                                  colAPS_Results_Flat,
                                                  ForAll(
                                                      Table(ParseJSON(SearchResults)),
                                                      {
                                                          PersonID: Coalesce(
                                                              Text(ThisRecord.Value.personId),
                                                              Text(ThisRecord.Value.PersonID)
                                                          ),
                                                          DisplayName: Coalesce(
                                                              Text(ThisRecord.Value.displayName),
                                                              Trim(
                                                                  Text(ThisRecord.Value.firstName) & " " &
                                                                  Text(ThisRecord.Value.lastName)
                                                              )
                                                          ),
                                                          DOB: If(
                                                              IsBlank(Text(ThisRecord.Value.dob)),
                                                              Blank(),
                                                              DateValue(Text(ThisRecord.Value.dob))
                                                          ),
                                                          SexLabel: Coalesce(
                                                              Text(ThisRecord.Value.sexLabel),
                                                              Text(ThisRecord.Value.gender)
                                                          ),
                                                          SSN: Text(ThisRecord.Value.ssn),
                                                          FullAddress: Coalesce(
                                                              Text(ThisRecord.Value.fullAddress),
                                                              "NA"
                                                          )
                                                      }
                                                  )
                                              );
                                              Clear(colAPS_Results);
                                              Clear(colPhoneNums);
                                              Clear(colAddressNums),
                                              Set(varAPS_Error, "Search service is temporarily unavailable. Please try again.");
                                              Set(varCanSearch, false);
                                              Clear(colAPS_Results);
                                              Clear(colAPS_Results_Flat);
                                              Clear(colPhoneNums);
                                              Clear(colAddressNums)
                                          );
                                          If(
                                              CountRows(colAPS_Results_Flat) = 0,
                                              Set(varNoResultsMessage, "No Matching Person Records Were Found in the System."),
                                              Set(varNoResultsMessage, "")
                                          )
                                      );
                                      
                                      Set(varIsSearching, false);
                                      
                                      // Set up Pagination
                                      If(
                                          varCanSearch,
                                          Set(PageSize, 16);
                                          Set(TotalPages, RoundUp(CountRows(colAPS_Results_Flat) / PageSize, 0))
                                      );
                                      
                                      
                                      
                                      // 3. Navigate to the Results page.
                                      If(
                                          varCanSearch,
                                          Navigate(PersonResultsScreen)
                                      );
                                    Text: ="Search"
                                    Width: =100
                              - btnClear:
                                  Control: FluentV8/Button@1.5.13
                                  Properties:
                                    AlignInContainer: =AlignInContainer.Center

                                      Reset(txtAddress1);
                                      Reset(txtAddress2);
                                      Reset(TextBox10);
                                      Reset(TextBox11);
                                       Reset(TextBox12);
                                       Reset(txtCounty);
                                       Reset(tglExactMatch);
                                       Reset(cmbGender);
                                       Reset(cmbRace);
                                       Set(varAPS_Error, "");
                                       Set(varNoResultsMessage, "");
                                       Set(varMatchMode, "FuzzyPhonetic");
                                       Set(varSearchClicked, false);
                                       Set(varIsLoading, false);
                                       Clear(colAPS_Results);
                                      Clear(colAPS_Results_Flat);
                                      Clear(colPhoneNums);
                                      Clear(colAddressNums);
                                    Text: ="Clear"
                                    VerticalAlignment: =VerticalAlign.Middle
                                    Width: =100
                  - Loading:
                      Control: FluentV8/Label@1.8.6
                      Properties:
                        AutoHeight: =true

                          =//varSearchClicked && !varIsLoading && CountRows(colAPS_Results_Flat) = 0 && IsBlank(varAPS_Error)
                          varIsSearching && varSearchClicked
                  - Error:
                      Control: FluentV8/Label@1.8.6
                      Properties:

                          //   Power Apps evaluates the Reset PROPERTY (not function)
                          //   on cmbGender and cmbRace declaratively — it resets the
                          //   control to its Default value without triggering OnChange.

/* =========================================================
   RESET STATE
   ========================================================= */
Set(varAPS_Error,    "");
Set(varNoResultsMessage, "");
Set(varIsSearching,  true);
Set(varCanSearch,    true);
Set(varIsValidDOB,   true);
Set(varSearchClicked, true);
Set(varMatchMode, If(tglExactMatch.Checked, "Exact", "FuzzyPhonetic"));
Clear(colAPS_Results);
Clear(colAPS_Results_Flat);
Clear(colPhoneNums);
Clear(colAddressNums);
Set(CurrentPage, 1);

/* =========================================================
   NORMALIZE INPUTS
   ========================================================= */
Set(var_fn,       Trim(txtFirstName.Value));
Set(var_ln,       Trim(txtLastName.Value));
Set(var_dobRaw,   Trim(txtDOB.Value));
Set(var_ssnRaw,   Trim(txtSSN.Value));
Set(var_pidRaw,   Trim(TextBox5.Value));
Set(var_phoneRaw, Trim(txtPhoneNUmber.Value));

Set(var_addr1,    Lower(Trim(txtAddress1.Value)));
Set(var_addr2,    Lower(Trim(txtAddress2.Value)));
Set(var_city,     Lower(Trim(TextBox10.Value)));
Set(var_state,    Lower(Trim(TextBox11.Value)));
Set(var_zip,      Lower(Trim(TextBox12.Value)));
Set(var_county,   Lower(Trim(txtCounty.Value)));

Set(var_gender, cmbGender.Selected);
Set(var_race,   cmbRace.SelectedItems);

Set(
    varSearchRequest,
    {
        matchMode: varMatchMode,
        firstName: var_fn,
        lastName: var_ln,
        dob: var_dobRaw,
        ssn: var_ssnRaw,
        personId: var_pidRaw,
        phone: var_phoneRaw,
        addressLine1: var_addr1,
        addressLine2: var_addr2,
        city: var_city,
        state: var_state,
        zipCode: var_zip,
        county: var_county,
        gender: Coalesce(var_gender.Value, ""),
        race: Concat(var_race, Value, ","),
        singleLineQuery: ""
    }
);


/* =========================================================
   DERIVED VALUES
   ========================================================= */
Set(var_pid,         If(!IsBlank(var_pidRaw), Value(var_pidRaw), Blank()));
Set(var_phoneDigits, Concat(MatchAll(var_phoneRaw, "\d"), FullMatch));
Set(var_ssn,         Substitute(Substitute(var_ssnRaw, "-", ""), " ", ""));
Set(var_dobClean,    Substitute(var_dobRaw, " ", ""));


/* =========================================================
   RULE 1 — PRESENCE CHECK
   ========================================================= */
If(
    varCanSearch             &&
    IsBlank(var_fn)          &&
    IsBlank(var_ln)          &&
    IsBlank(var_dobRaw)      &&
    IsBlank(var_ssnRaw)      &&
    IsBlank(var_pidRaw)      &&
    IsBlank(var_phoneRaw)    &&
    IsBlank(var_addr1)       &&
    IsBlank(var_addr2)       &&
    IsBlank(var_city)        &&
    IsBlank(var_state)       &&
    IsBlank(var_zip)         &&
    IsBlank(var_county)      &&
    IsBlank(var_gender.Value)&&
    CountRows(var_race) = 0,
    Set(varAPS_Error, "Enter at least one valid search identifier.");
    Set(varCanSearch, false)
);


/* =========================================================
   FIELD-LEVEL VALIDATIONS
   ========================================================= */
If(
    varCanSearch && !IsBlank(var_fn) &&
    !IsMatch(var_fn, "^[A-Za-z' -]+$"),
    Set(varAPS_Error, "Invalid First Name.");
    Set(varCanSearch, false)
);
If(
    varCanSearch && !IsBlank(var_ln) &&
    !IsMatch(var_ln, "^[A-Za-z' -]+$"),
    Set(varAPS_Error, "Invalid Last Name.");
    Set(varCanSearch, false)
);
If(
    varCanSearch && !IsBlank(var_pidRaw) &&
    (!IsMatch(var_pidRaw, "^\d+$") || Len(var_pidRaw) < 2),
    Set(varAPS_Error, "Person ID must be numeric and at least 2 digits.");
    Set(varCanSearch, false)
);
If(
    varCanSearch && !IsBlank(var_phoneRaw) &&
    !IsMatch(var_phoneRaw, "^[\d\s\(\)\-\+\.x]+$"),
    Set(varAPS_Error, "Invalid Phone Number.");
    Set(varCanSearch, false)
);
If(
    varCanSearch && !IsBlank(var_ssnRaw) &&
    !(Len(var_ssn) = 9 || Len(var_ssn) = 4),
    Set(varAPS_Error, "SSN must be 9 or last 4 digits.");
    Set(varCanSearch, false)
);


/* =========================================================
   DOB VALIDATION
   ========================================================= */
Set(
    varIsValidDOB,
    IsBlank(var_dobRaw) ||
    IsMatch(var_dobClean, "^\d{4}$") ||
    IsMatch(var_dobClean, "^(0?[1-9]|1[0-2])/\d{4}$") ||
    (
        IsMatch(var_dobClean, "^(0?[1-9]|1[0-2])/(0?[1-9]|[12]\d|3[01])/\d{4}$") &&
        With(
            { dobParts: Split(var_dobClean, "/") },
            With(
                {
                    m: Value(First(dobParts).Value),
                    d: Value(Index(dobParts, 2).Value),
                    y: Value(Last(dobParts).Value)
                },
                Month(Date(y, m, d)) = m &&
                Day(Date(y, m, d))   = d
            )
        )
    )
);
If(
    varCanSearch && !varIsValidDOB,
    Set(varAPS_Error, "DOB must be MM/DD/YYYY, MM/YYYY, or YYYY.");
    Set(varCanSearch, false)
);


/* =========================================================
   MAIN SEARCH
   ========================================================= */
If(
    varCanSearch,
    IfError(
        Set(
            SearchResults,
            PersonSearchDataFlow.Run(
                JSON(varSearchRequest, JSONFormat.Compact)
            ).results
        );
        ClearCollect(
            colAPS_Results_Flat,
            ForAll(
                Table(ParseJSON(SearchResults)),
                {
                    PersonID: Coalesce(
                        Text(ThisRecord.Value.personId),
                        Text(ThisRecord.Value.PersonID)
                    ),
                    DisplayName: Coalesce(
                        Text(ThisRecord.Value.displayName),
                        Trim(
                            Text(ThisRecord.Value.firstName) & " " &
                            Text(ThisRecord.Value.lastName)
                        )
                    ),
                    DOB: If(
                        IsBlank(Text(ThisRecord.Value.dob)),
                        Blank(),
                        DateValue(Text(ThisRecord.Value.dob))
                    ),
                    SexLabel: Coalesce(
                        Text(ThisRecord.Value.sexLabel),
                        Text(ThisRecord.Value.gender)
                    ),
                    SSN: Text(ThisRecord.Value.ssn),
                    FullAddress: Coalesce(
                        Text(ThisRecord.Value.fullAddress),
                        "NA"
                    )
                }
            )
        );
        Clear(colAPS_Results);
        Clear(colPhoneNums);
        Clear(colAddressNums),
        Set(varAPS_Error, "Search service is temporarily unavailable. Please try again.");
        Set(varCanSearch, false);
        Clear(colAPS_Results);
        Clear(colAPS_Results_Flat);
        Clear(colPhoneNums);
        Clear(colAddressNums)
    );
    If(
        CountRows(colAPS_Results_Flat) = 0,
        Set(varNoResultsMessage, "No Matching Person Records Were Found in the System."),
        Set(varNoResultsMessage, "")
    )
);

Set(varIsSearching, false);

// Set up Pagination
If(
    varCanSearch,
    Set(PageSize, 16);
    Set(TotalPages, RoundUp(CountRows(colAPS_Results_Flat) / PageSize, 0))
);



// 3. Navigate to the Results page.
If(
    varCanSearch,
    Navigate(PersonResultsScreen)
);
