         Person_PersonAddress is a lookup column.  The  in
         operator on lookup columns is not delegable to
         Dataverse, causing the delegation warning and
         returning empty results beyond the first 500 rows.
    ===================================================== */
    If(
        !IsBlank(var_addr1) || !IsBlank(var_addr2) ||
        !IsBlank(var_city)  || !IsBlank(var_state)  ||
        !IsBlank(var_zip)   || !IsBlank(var_county),

        Clear(colAddressNums);
        ForAll(
            colAPS_Results As p,
            Collect(
                colAddressNums,
                Filter(
                    'Person Addresses',
                    Person_PersonAddress.'Person ID' = p.'Person ID' &&
                    statecode = 0
                )
            )
        );

        ClearCollect(
            colAPS_Results,
            Filter(
                colAPS_Results As p,
                CountRows(
                    Filter(
                        colAddressNums As a,
                        a.Person_PersonAddress.'Person ID' = p.'Person ID' &&
                        (IsBlank(var_addr1)  || Find(var_addr1,  Lower(Coalesce(a.cw_addressline1, ""))) > 0) &&
                        (IsBlank(var_addr2)  || Find(var_addr2,  Lower(Coalesce(a.cw_addressline2, ""))) > 0) &&
                        (IsBlank(var_city)   || Find(var_city,   Lower(Coalesce(a.cw_city,         ""))) > 0) &&
                        (IsBlank(var_state)  || Find(var_state,  Lower(Coalesce(a.cw_state,        ""))) > 0) &&
                        (IsBlank(var_zip)    || Find(var_zip,    Lower(Coalesce(a.cw_zipcode,      ""))) > 0) &&
                        (IsBlank(var_county) || Find(var_county, Lower(Coalesce(a.cw_county,       ""))) > 0)
                    )
                ) > 0
            )
        )
    );


    /* =====================================================
       STEP 9 — FINAL FLATTEN
    ===================================================== */
    Clear(colAPS_Results_Flat);

    ForAll(
        colAPS_Results As p,
        With(
            {
                displayName:
                    Trim(
                        Coalesce(Text(p.'First Name'), "") & " " &
                        Coalesce(Text(p.'Last Name'),  "")
                    ),
                addrs:
                    If(
                        !IsBlank(var_addr1) || !IsBlank(var_addr2) ||
                        !IsBlank(var_city)  || !IsBlank(var_state)  ||
                        !IsBlank(var_zip)   || !IsBlank(var_county),

                        // Address searched → only matching rows (already in colAddressNums)
                        Filter(
                            colAddressNums As a,
                            a.Person_PersonAddress.'Person ID' = p.'Person ID' &&
                            (IsBlank(var_addr1)  || Find(var_addr1,  Lower(Coalesce(a.cw_addressline1, ""))) > 0) &&
                            (IsBlank(var_addr2)  || Find(var_addr2,  Lower(Coalesce(a.cw_addressline2, ""))) > 0) &&
                            (IsBlank(var_city)   || Find(var_city,   Lower(Coalesce(a.cw_city,         ""))) > 0) &&
                            (IsBlank(var_state)  || Find(var_state,  Lower(Coalesce(a.cw_state,        ""))) > 0) &&
                            (IsBlank(var_zip)    || Find(var_zip,    Lower(Coalesce(a.cw_zipcode,      ""))) > 0) &&
                            (IsBlank(var_county) || Find(var_county, Lower(Coalesce(a.cw_county,       ""))) > 0)
                        ),

                        // No address searched → all active addresses
                        Filter(
                            'Person Addresses',
                            Person_PersonAddress.'Person ID' = p.'Person ID' &&
                            statecode = 0
                        )
                    )
            },

            If(
                CountRows(addrs) = 0,
                Collect(
                    colAPS_Results_Flat,
                    {
                        PersonID:    p.'Person ID',
                        DisplayName: displayName,
                        DOB:         p.'Date of Birth',
                        SexLabel:    Coalesce(Text((p.Sex),"")),
                        SSN:         Coalesce(p.SSN,""),
                        FullAddress: "NA"
                    }
                ),
                ForAll(
                    addrs As a,
                    Collect(
                        colAPS_Results_Flat,
                        {
                            PersonID:    p.'Person ID',
                            DisplayName: displayName,
                            DOB:         p.'Date of Birth',
                            SSN:        Coalesce(p.SSN,""),  //"SSN : " & Coalesce(ThisItem.SSN, "NA")
                            SexLabel:    Coalesce(Text((p.Sex),"")),
                            FullAddress:
                                Trim(
                                    Coalesce(a.cw_addressline1, "") &
                                    If(!IsBlank(a.cw_addressline2), ", " & a.cw_addressline2, "") &
                                    If(!IsBlank(a.cw_city),        ", " & a.cw_city,          "") &
                                    If(!IsBlank(a.cw_state),       ", " & a.cw_state,          "") &
                                    If(!IsBlank(a.cw_zipcode),     " "  & a.cw_zipcode,        "") &
                                    If(!IsBlank(a.cw_county),      ", " & a.cw_county,         "")
                                )
                        }
                    )
                )
            )
        )
    );

    If(
        CountRows(colAPS_Results_Flat) = 0,
        Set(varAPS_Error, "No records found. Refine your search criteria."),
        Set(varAPS_Error, "")
    )
);

Set(varIsSearching, false);



// Set up Pagination
Set(PageSize, 16);
